void adj_list2yale_ars(int V, adj_l_el *adj_l_mat[V], int *v_ar[V], int **e_ar){
    int i,j=0,k=0, size=V*V;
    adj_l_el *tmp;
    int *edge_ar2=(int*)malloc(size*sizeof(int));
    for(i=0;i<V;i++){
        tmp=adj_l_mat[i];
        v_ar[i]=k;
        while(tmp){
            if(k>=size)
                edge_ar2=realloc(edge_ar2, (size=size*2)*sizeof(int));
            edge_ar2[k++]=tmp->v;
            tmp=tmp->next;
        }
    }
    e_ar=realloc(edge_ar2, k*sizeof(int));
    return;
}
